FUNCTION(GENERATE_OPENHRP_SCRIPTS)
  MATH(EXPR END ${ARGC}-1)
  FOREACH(I RANGE 0 ${END})
    SET(SCRIPT ${ARGV${I}})
    # Generate script.
    CONFIGURE_FILE(${SCRIPT}.cmake ${CMAKE_CURRENT_BINARY_DIR}/${SCRIPT}.py)
  ENDFOREACH(I RANGE 0 ${END})
ENDFUNCTION(GENERATE_OPENHRP_SCRIPTS)

GENERATE_OPENHRP_SCRIPTS(
  hrp_configs
  sot_minimal
)
